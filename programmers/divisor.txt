def solution(left, right):
    answer = 0
    
    for i in range(left, (right+1)):
        if(NumberOfDivisor(i) % 2 == 0) :
            answer += i
        else : 
            answer -= i
    
    return answer

def NumberOfDivisor(n):

    divisors = []

    for i in range(1, int(n**(1/2)) + 1) :
        if(n % i == 0) :
            divisors.append(i)
            if((i**2) != n) :
                divisors.append(n // i)
    #divisors.sort()
    return len(divisors)
